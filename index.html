uniform float time; void main() { // This function gets 'called' for each pixel in the image // set constants using const. Almost everything is a float, // don't forget to add the .0 at the end, else stuff breaks const float x_scale = 6.0; const float y_scale = 100.0; const float time_scaler = 20.0; // use gl_FragCoord to access the current coordinate float hue_1 = (gl_FragCoord.y + time * time_scaler) / x_scale; float hue_2 = gl_FragCoord.x / y_scale; vec3 rgb = hsl2rgb(mod(hue_1 + hue_2, 12.0) / 12.0, 1.0, 0.5); // gl_FragColor is a special variable in a fragment shader // it is responsible for setting the color of the pixel gl_FragColor = vec4(rgb, 1.0); }
